Definition hzdiv0 : hz → hz → hz → UU :=
  fun n m k ⇒ n × k = m.

Definition hzdiv : hrel hz := fun n m ⇒ ∃ k : hz, hzdiv0 n m k.

Lemma hzdivisrefl : isrefl hzdiv.
Proof.
  red.
  intro.
  unfold hzdiv.
  apply total2tohexists.
  split with 1.
  apply hzmultr1.
Defined.

Lemma hzdivistrans : istrans hzdiv.
Proof.
  intros a b c p q.
  use (hinhuniv _ p).
  intro k.
  destruct k as [ k f ].
  use (hinhuniv _ q).
  intro l.
  destruct l as [ l g ].
  intros P s.
  apply s.
  unfold hzdiv0 in f, g.
  split with ( k × l ).
  red.
  rewrite <- hzmultassoc.
  rewrite f.
  assumption.
Defined.
